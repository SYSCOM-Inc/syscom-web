asm-migration.yml 
# ASM Migration Configuration
migration:
  name: "FileNet to SharePoint"
  source:
    type: filenet-p8
    object_store: "Production_OS"
    query: "SELECT * FROM Document"
  target:
    type: sharepoint-online
    site: "enterprise-docs"
    library: "Migrated Content"
  transforms:
    - type: tiff-to-pdf
      ocr: true
      full_text: true
    - type: metadata-map
      mode: ai-auto  # AI-powered mapping
  options:
    parallel_threads: 16
    preserve_security: true
    preserve_versions: true
    anomaly_detection: true
